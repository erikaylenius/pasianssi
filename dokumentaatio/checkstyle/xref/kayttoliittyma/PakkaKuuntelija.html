1   package kayttoliittyma;

3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import javax.swing.JLayeredPane;
6   import java.awt.Container;
7   
8   /**
9    * Luokka kuuntelee pakkaan kohdistuvia hiiren klikkauksia ja huolehtii seuraavan 
10   * kortin näyttämisestä.
11   */
12  
13  public class PakkaKuuntelija implements ActionListener {
14  
15      public JLayeredPane nakyvaPakka;
16      public Container container;
17      
18      /**
19       * Konstruktori.
20       * @param container pakan sisältävä container
21       * @param nakyvaPakka pakkaa visualisoiva JLabelKortti-pino
22       */
23      public PakkaKuuntelija(Container container, JLayeredPane nakyvaPakka) {
24          this.container = container;
25          this.nakyvaPakka = nakyvaPakka;
26      }
27      
28      /**
29       * Metodi siirtää pakan päällimmäisen kortin alimmaiseksi.
30       * @param ae hiiren klikkaus
31       */
32  
33      @Override
34      public void actionPerformed(ActionEvent ae) {       
35          
36  
37          JLabelKortti komponentti = (JLabelKortti) this.nakyvaPakka.getComponent(0);
38          this.nakyvaPakka.setLayer(komponentti, -1);
39  
40      }
41      
42  }
