3   /**
4    * Luokka määrittelee seitsemän eri pelin alussa jaetun pinon ominaisuudet. 
5    * Pinot sisältävät aluksi 1, 2, 3, 4, 5, 6 ja 7 arvottua korttia. Alkuasetelmassa 
6    * kunkin pinon ylin kortti on oikein päin, loput nurinpäin. 
7    */
8   
9   import java.util.ArrayList;
10  
11  public class Poytapino {
12  
13      private ArrayList<Kortti> poytaPino;
14  
15      public Poytapino() {
16          
17          this.poytaPino = new ArrayList<Kortti>();
18          
19      } 
20      
21      /**
22       * Metodi lisää kortin pöytäpinoon uuden pelin alkaessa.
23       * @param kortti pinoon lisättävä kortti
24       */
25      
26      public void lisaaKortti(Kortti kortti) {
27          this.poytaPino.add(kortti);
28      }
29      
30      /**
31       * Metodi huolehtii siitä, että sääntöjen mukaiset kortit ovat pelin alkaessa
32       * ylösalaisin.
33       */
34      
35      public void alusta() {
36          for (int i = 0; i < (poytaPino.size() - 2); i++) {
37              poytaPino.get(i).kaanna();
38          }
39      }
40      
41      public int pinonKoko() {
42          return this.poytaPino.size();
43      }
44      
45      public Kortti getKortti(int i) {
46          return this.poytaPino.get(i);
47      }
48      
49      /**
50       * Metodi lisää pelin aikana pinon päälle halutun kortin, mikäli säännöt 
51       * sallivat kortin lisäämisen.
52       * @param kortti kortti, joka halutaan lisätä
53       */
54      
55      public void lisaaKorttiPaalle(Kortti kortti) {
56          if (poytaPino.isEmpty()) {
57              if (kortti.getArvo() == 13) {
58                  poytaPino.add(kortti);
59              }
60          } else {
61              if (poytaPino.get(poytaPino.size() - 1).onkoOikeinPain()) {
62                  if (!poytaPino.get(poytaPino.size() - 1).getVari().equals(kortti.getVari())) {
63                      if ((poytaPino.get(poytaPino.size() - 1).getArvo()) == (kortti.getArvo() + 1)) {
64                          poytaPino.get(poytaPino.size() - 1).kaanna();
65                          poytaPino.add(kortti);
66                      }
67                  } 
68              }
69          }    
70      }
71      
72      /**
73       * Metodi poistaa pinon päällimmäisen kortin pinosta silloin, kun kyseinen
74       * kortti voidaan sääntöjen mukaan siirtää muualle.
75       * @param kortti kortti, joka siirretään muualle
76       */
77  
78      public void poistaKorttiPinosta(Kortti kortti) {
79          this.poytaPino.remove(kortti);
80      }
81  }
